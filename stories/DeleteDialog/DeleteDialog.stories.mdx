```jsx 
	import { Deletedialog } from '@views_components'


	...
	const [deleteDialogVisible, setDeleteDialogVisible] = useState(false)
	...

	const handleAgree = () => {
		// do something when hit button agree
	}

	return (
		...
		<DeleteDialog
			open={deleteDialogVisible}
			onClose={() => {
				setDeleteDialogVisible(false)
			}}
			onAgree={() => {
				setDeleteDialogVisible(false)
				handleAgree()
			}}
			onDisagree={() => {
				setDeleteDialogVisible(false)
			}}
		/>
		...
	)
```
